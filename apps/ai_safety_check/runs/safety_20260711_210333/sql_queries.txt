-- advisories:mlflow
SELECT DISTINCT
    a."Title",
    a."CVSS3Score",
    a."GitHubSeverity"
FROM "DEPS_DEV_V1"."DEPS_DEV_V1"."PACKAGEVERSIONS" pv,
     LATERAL FLATTEN(INPUT => pv."Advisories") f
JOIN "DEPS_DEV_V1"."DEPS_DEV_V1"."ADVISORIES" a
  ON a."SourceID" = f.VALUE:"SourceID"::VARCHAR
WHERE pv."Name" = 'mlflow'
ORDER BY a."CVSS3Score" DESC NULLS LAST
