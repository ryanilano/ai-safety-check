WITH requested_systems AS (
  SELECT column1 AS "System"
  FROM VALUES ('PYPI'),('NPM'),('MAVEN'),('CARGO'),('GO'),('NUGET')
)
SELECT rs."System",
       COALESCE(pv."VersionCount", 0) AS "VersionCount"
FROM requested_systems rs
LEFT JOIN (
    SELECT "System", COUNT(*) AS "VersionCount"
    FROM DEPS_DEV_V1.DEPS_DEV_V1."PACKAGEVERSIONS"
    WHERE "Name" = 'gradio'
    GROUP BY "System"
) pv ON rs."System" = pv."System"
ORDER BY "VersionCount" DESC, rs."System" ASC
